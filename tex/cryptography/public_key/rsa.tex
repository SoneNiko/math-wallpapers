\begin{wallpaperframe}
Let $n = pq$ for distinct primes $p, q$, and let $e, d$ satisfy $e \cdot d \equiv 1 \pmod{\phi(n)}$.
\begin{align*}
\mathcal{E}_e &: \mathbb{Z}_n \to \mathbb{Z}_n, \quad m \mapsto m^e \bmod n \\
\mathcal{D}_d &: \mathbb{Z}_n \to \mathbb{Z}_n, \quad c \mapsto c^d \bmod n
\end{align*}
\[
\mathcal{D}_d \circ \mathcal{E}_e = \mathrm{id}_{\mathbb{Z}_n}
\]
\end{wallpaperframe}
